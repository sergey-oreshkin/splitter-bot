user
    (PK id, firs_name, last_name, nic, updated(current timestamp))

chat

user_chat
    (PK id, FK user(user.id), FK chat(chat.id))

split
    (PK id, FK owner(user_chat.id))

expense(split_record)
    (PK id, FK split(split.id), FK who_paid(user.id), share JSONb)
